        function drawRoad() {
            // Φόντο - ουρανός με gradient
            const skyGradient = ctx.createLinearGradient(0, 0, 0, canvas.height);
            skyGradient.addColorStop(0, '#87CEEB');
            skyGradient.addColorStop(1, '#98D8E8');
            ctx.fillStyle = skyGradient;
            ctx.fillRect(0, 0, canvas.width, canvas.height);
            
            // Χλοοτάπητας με gradient
            const grassGradient = ctx.createLinearGradient(0, 0, 0, canvas.height);
            grassGradient.addColorStop(0, '#2ecc71');
            grassGradient.addColorStop(1, '#27ae60');
            ctx.fillStyle = grassGradient;
            ctx.fillRect(0, 0, 100, canvas.height);
            ctx.fillRect(500, 0, 100, canvas.height);
            
            // Άκρες δρόμου (πεζοδρόμιο)
            ctx.fillStyle = '#95a5a6';
            ctx.fillRect(95, 0, 10, canvas.height);
            ctx.fillRect(495, 0, 10, canvas.height);
            
            // Δρόμος με gradient
            const roadGradient = ctx.createLinearGradient(100, 0, 500, 0);
            roadGradient.addColorStop(0, '#2c3e50');
            roadGradient.addColorStop(0.5, '#34495e');
            roadGradient.addColorStop(1, '#2c3e50');
            ctx.fillStyle = roadGradient;
            ctx.fillRect(105, 0, 390, canvas.height);
            
            // Σκιές στις άκρες του δρόμου
            ctx.fillStyle = 'rgba(0,0,0,0.2)';
            ctx.fillRect(105, 0, 20, canvas.height);
            ctx.fillRect(475, 0, 20, canvas.height);
            
            // Γραμμές δρόμου με κίνηση
            ctx.strokeStyle = '#f39c12';
            ctx.lineWidth = 5;
            ctx.setLineDash([30, 20]);
            ctx.shadowBlur = 5;
            ctx.shadowColor = 'rgba(243, 156, 18, 0.5)';
            
            for (let i = 0; i < 2; i++) {
                ctx.beginPath();
                ctx.moveTo(225 + i * 125, -roadOffset);
                ctx.lineTo(225 + i * 125, canvas.height - roadOffset);
                ctx.stroke();
            }
            
            ctx.setLineDash([]);
            ctx.shadowBlur = 0;
            
            // Θάμνοι αριστερά
            bushes.forEach(bush => {
                if (bush.side === 'left') {
                    // Σκιά θάμνου
                    ctx.fillStyle = 'rgba(0,0,0,0.3)';
                    ctx.beginPath();
                    ctx.ellipse(bush.x + 5, bush.y + bush.size + 5, bush.size, bush.size * 0.3, 0, 0, Math.PI * 2);
                    ctx.fill();
                    
                    // Θάμνος με gradient
                    const bushGradient = ctx.createRadialGradient(bush.x, bush.y - bush.size/4, bush.size/4, bush.x, bush.y, bush.size);
                    bushGradient.addColorStop(0, '#2ecc71');
                    bushGradient.addColorStop(1, '#27ae60');
                    ctx.fillStyle = bushGradient;
                    ctx.beginPath();
                    ctx.arc(bush.x, bush.y, bush.size, 0, Math.PI * 2);
                    ctx.fill();
                    
                    ctx.fillStyle = '#229954';
                    ctx.beginPath();
                    ctx.arc(bush.x - bush.size/3, bush.y, bush.size * 0.7, 0, Math.PI * 2);
                    ctx.fill();
                    ctx.beginPath();
                    ctx.arc(bush.x + bush.size/3, bush.y, bush.size * 0.7, 0, Math.PI * 2);
                    ctx.fill();
                    
                    // Highlights
                    ctx.fillStyle = 'rgba(255,255,255,0.3)';
                    ctx.beginPath();
                    ctx.arc(bush.x - bush.size/5, bush.y - bush.size/5, bush.size * 0.3, 0, Math.PI * 2);
                    ctx.fill();
                }
            });
            
            // Θάμνοι δεξιά
            bushes.forEach(bush => {
                if (bush.side === 'right') {
                    // Σκιά θάμνου
                    ctx.fillStyle = 'rgba(0,0,0,0.3)';
                    ctx.beginPath();
                    ctx.ellipse(bush.x + 5, bush.y + bush.size + 5, bush.size, bush.size * 0.3, 0, 0, Math.PI * 2);
                    ctx.fill();
                    
                    // Θάμνος με gradient
                    const bushGradient = ctx.createRadialGradient(bush.x, bush.y - bush.size/4, bush.size/4, bush.x, bush.y, bush.size);
                    bushGradient.addColorStop(0, '#2ecc71');
                    bushGradient.addColorStop(1, '#27ae60');
                    ctx.fillStyle = bushGradient;
                    ctx.beginPath();
                    ctx.arc(bush.x, bush.y, bush.size, 0, Math.PI * 2);
                    ctx.fill();
                    
                    ctx.fillStyle = '#229954';
                    ctx.beginPath();
                    ctx.arc(bush.x - bush.size/3, bush.y, bush.size * 0.7, 0, Math.PI * 2);
                    ctx.fill();
                    ctx.beginPath();
                    ctx.arc(bush.x + bush.size/3, bush.y, bush.size * 0.7, 0, Math.PI * 2);
                    ctx.fill();
                    
                    // Highlights
                    ctx.fillStyle = 'rgba(255,255,255,0.3)';
                    ctx.beginPath();
                    ctx.arc(bush.x - bush.size/5, bush.y - bush.size/5, bush.size * 0.3, 0, Math.PI * 2);
                    ctx.fill();
                }
            });
        }

        function drawCar(x, y, width, height, color) {
            // Κύριο σώμα αυτοκινήτου με gradient
            const bodyGradient = ctx.createLinearGradient(x, y, x + width, y);
            bodyGradient.addColorStop(0, color);
            bodyGradient.addColorStop(0.5, color);
            bodyGradient.addColorStop(1, shadeColor(color, -20));
            ctx.fillStyle = bodyGradient;
            ctx.fillRect(x + 5, y + 20, width - 10, height - 30);
            
            // Οροφή (cabin) με gradient
            const cabinGradient = ctx.createLinearGradient(x, y + 25, x, y + 50);
            cabinGradient.addColorStop(0, shadeColor(color, -30));
            cabinGradient.addColorStop(1, color);
            ctx.fillStyle = cabinGradient;
            ctx.fillRect(x + 8, y + 25, width - 16, 25);
            
            // Παρμπρίζ μπροστά με reflection
            const windshieldGradient = ctx.createLinearGradient(x, y + 15, x, y + 25);
            windshieldGradient.addColorStop(0, '#4a5f7f');
            windshieldGradient.addColorStop(1, '#1a1a2e');
            ctx.fillStyle = windshieldGradient;
            ctx.beginPath();
            ctx.moveTo(x + 8, y + 25);
            ctx.lineTo(x + 12, y + 15);
            ctx.lineTo(x + width - 12, y + 15);
            ctx.lineTo(x + width - 8, y + 25);
            ctx.closePath();
            ctx.fill();
            
            // Reflection στο παρμπρίζ
            ctx.fillStyle = 'rgba(135, 206, 235, 0.3)';
            ctx.beginPath();
            ctx.moveTo(x + 8, y + 25);
            ctx.lineTo(x + 12, y + 15);
            ctx.lineTo(x + width - 12, y + 15);
            ctx.lineTo(x + 8, y + 25);
            ctx.closePath();
            ctx.fill();
            
            // Παρμπρίζ πίσω
            ctx.fillStyle = windshieldGradient;
            ctx.beginPath();
            ctx.moveTo(x + 8, y + 50);
            ctx.lineTo(x + 12, y + 60);
            ctx.lineTo(x + width - 12, y + 60);
            ctx.lineTo(x + width - 8, y + 50);
            ctx.closePath();
            ctx.fill();
            
            // Προφυλακτήρας μπροστά
            ctx.fillStyle = '#2c3e50';
            ctx.fillRect(x + 5, y + 10, width - 10, 8);
            ctx.fillStyle = 'rgba(255,255,255,0.2)';
            ctx.fillRect(x + 5, y + 10, width - 10, 2);
            
            // Προφυλακτήρας πίσω
            ctx.fillStyle = '#2c3e50';
            ctx.fillRect(x + 5, y + height - 15, width - 10, 8);
            
            // Φώτα μπροστά με glow

            ctx.fillRect(x + 8, y + height - 12, 8, 4);

        // Helper function για να κάνουμε σκούρα/ανοιχτά χρώματα
        function shadeColor(color, percent) {
            const num = parseInt(color.replace("#",""), 16);
            const amt = Math.round(2.55 * percent);
            const R = (num >> 16) + amt;
            const G = (num >> 8 & 0x00FF) + amt;
            const B = (num & 0x0000FF) + amt;
            return "#" + (0x1000000 + (R<255?R<1?0:R:255)*0x10000 +
                (G<255?G<1?0:G:255)*0x100 + (B<255?B<1?0:B:255))
                .toString(16).slice(1);
        }

        function drawTruck(x, y, width, height, color) {
            // Καμπίνα φορτηγού
            const cabinGradient = ctx.createLinearGradient(x, y, x + width, y);
            cabinGradient.addColorStop(0, color);
            cabinGradient.addColorStop(1, shadeColor(color, -20));
            ctx.fillStyle = cabinGradient;
            ctx.fillRect(x + 5, y + 10, width - 10, 30);
            
            // Παρμπρίζ
            ctx.fillStyle = '#1a1a2e';
            ctx.fillRect(x + 8, y + 15, width - 16, 15);
            
            // Κοντέινερ πίσω
            ctx.fillStyle = shadeColor(color, -30);
            ctx.fillRect(x + 5, y + 42, width - 10, height - 50);
            
            // Γραμμές στο κοντέινερ
            ctx.strokeStyle = shadeColor(color, -40);
            ctx.lineWidth = 2;
            for (let i = 1; i < 4; i++) {
                ctx.beginPath();
                ctx.moveTo(x + 5, y + 42 + i * 15);
                ctx.lineTo(x + width - 5, y + 42 + i * 15);
                ctx.stroke();
            }
            
            // Φώτα
            ctx.fillStyle = '#f1c40f';
            ctx.fillRect(x + 8, y + 12, 6, 3);
            ctx.fillRect(x + width - 14, y + 12, 6, 3);
            
            // Ρόδες
            ctx.fillStyle = '#1a1a1a';
            ctx.fillRect(x - 2, y + 25, 6, 12);
            ctx.fillRect(x + width - 4, y + 25, 6, 12);
            ctx.fillRect(x - 2, y + height - 15, 6, 12);
            ctx.fillRect(x + width - 4, y + height - 15, 6, 12);
        }

        function drawBus(x, y, width, height, color) {
            // Σώμα λεωφορείου
            const busGradient = ctx.createLinearGradient(x, y, x + width, y);
            busGradient.addColorStop(0, color);
            busGradient.addColorStop(1, shadeColor(color, -20));
            ctx.fillStyle = busGradient;
            ctx.fillRect(x + 5, y + 15, width - 10, height - 25);
            
            // Παράθυρα
            ctx.fillStyle = '#4a5f7f';
            const windowHeight = 12;
            const windowSpacing = 18;
            for (let i = 0; i < Math.floor((height - 30) / windowSpacing); i++) {
                ctx.fillRect(x + 8, y + 20 + i * windowSpacing, width - 16, windowHeight);
            }
            
            // Ρίγα λεωφορείου
            ctx.fillStyle = shadeColor(color, 30);
            ctx.fillRect(x + 5, y + height/2, width - 10, 4);
            
            // Φώτα μπροστά
            ctx.fillStyle = '#f1c40f';
            ctx.fillRect(x + 8, y + 12, 8, 4);
            ctx.fillRect(x + width - 16, y + 12, 8, 4);
            
            // Φώτα πίσω
            ctx.fillStyle = '#e74c3c';
            ctx.fillRect(x + 8, y + height - 12, 8, 4);
            ctx.fillRect(x + width - 16, y + height - 12, 8, 4);
            
            // Ρόδες
            ctx.fillStyle = '#1a1a1a';
            ctx.fillRect(x - 2, y + 20, 6, 12);
            ctx.fillRect(x + width - 4, y + 20, 6, 12);
            ctx.fillRect(x - 2, y + height - 20, 6, 12);
            ctx.fillRect(x + width - 4, y + height - 20, 6, 12);
        }

        function spawnObstacle() {
            if (Math.random() < 0.008) {
                // Βρίσκουμε ποιες λωρίδες έχουν ήδη αυτοκίνητα κοντά
                const occupiedLanes = [];
                obstacles.forEach(obs => {
                    if (obs.y < 300) {
                        const laneIndex = lanePositions.indexOf(obs.x);
                        if (!occupiedLanes.includes(laneIndex)) {
                            occupiedLanes.push(laneIndex);
                        }
                    }
                });
                
                if (occupiedLanes.length >= 2) {
                    return;
                }
                
                const availableLanes = [0, 1, 2].filter(lane => !occupiedLanes.includes(lane));
                const lane = availableLanes[Math.floor(Math.random() * availableLanes.length)];
                
                // Διαφορετικοί τύποι οχημάτων
                const vehicleTypes = [
                    { type: 'small', width: 40, height: 60, colors: ['#e74c3c', '#3498db', '#2ecc71', '#f39c12', '#9b59b6', '#e67e22', '#1abc9c'] },
                    { type: 'car', width: 50, height: 80, colors: ['#c0392b', '#2980b9', '#27ae60', '#d68910', '#8e44ad', '#d35400', '#16a085'] },
                    { type: 'large', width: 55, height: 90, colors: ['#922b21', '#1f618d', '#1e8449', '#b9770e', '#6c3483'] },
                    { type: 'truck', width: 52, height: 100, colors: ['#7f8c8d', '#34495e', '#95a5a6'] },
                    { type: 'bus', width: 50, height: 110, colors: ['#f39c12', '#e67e22', '#d35400'] }
                ];
                
                // Διαλέγουμε τυχαίο τύπο οχήματος
                const vehicleType = vehicleTypes[Math.floor(Math.random() * vehicleTypes.length)];
                const color = vehicleType.colors[Math.floor(Math.random() * vehicleType.colors.length)];
                const speedMultiplier = 0.7 + Math.random() * 0.6;
                
                obstacles.push({
                    x: lanePositions[lane],
                    y: -120,
                    width: vehicleType.width,
                    height: vehicleType.height,
                    color: color,
                    type: vehicleType.type,
                    speedMultiplier: speedMultiplier
                });
            }
        }

        function checkCollision() {
            for (let obs of obstacles) {
                // Αυξημένο margin για ακόμα πιο ακριβές collision
                const margin = 12;
                if (player.x + margin < obs.x + obs.width - margin &&
                    player.x + player.width - margin > obs.x + margin &&
                    player.y + margin < obs.y + obs.height - margin &&
                    player.y + player.height - margin > obs.y + margin) {
                    return true;
                }
            }
            return false;
        }

        function gameLoop() {
            if (!gameRunning) return;

            ctx.clearRect(0, 0, canvas.width, canvas.height);
            
            // Ενημέρωση offset για κίνηση δρόμου
            roadOffset += speed / 10;
            if (roadOffset > 50) roadOffset = 0;
            
            // Κίνηση θάμνων
            bushes.forEach(bush => {
                bush.y += speed / 8;
                if (bush.y > canvas.height + 50) {
                    bush.y = -50;
                    bush.x = bush.side === 'left' ? 40 + Math.random() * 40 : 520 + Math.random() * 40;
                    bush.size = 20 + Math.random() * 15;
                }
            });
            
            drawRoad();

            distance += speed / 3.6 / 60;
            
            // Check for speed increase
            const settings = difficultySettings[difficulty];
            if (Math.floor(distance / settings.increaseInterval) > lastSpeedIncrease) {
                lastSpeedIncrease = Math.floor(distance / settings.increaseInterval);
                speed += settings.speedIncrease;
                showSpeedBoost(settings.speedIncrease);
            }
            
            currentTime = Math.floor((Date.now() - startTime) / 1000);
            const minutes = Math.floor(currentTime / 60);
            const seconds = currentTime % 60;
            document.getElementById('time').textContent = `${minutes}:${seconds.toString().padStart(2, '0')}`;
            document.getElementById('speed').textContent = Math.round(speed);
            document.getElementById('distance').textContent = Math.round(distance);

            if (keys['ArrowLeft'] && player.lane > 0) {
                player.lane--;
                player.x = lanePositions[player.lane];
                keys['ArrowLeft'] = false;
            }
            if (keys['ArrowRight'] && player.lane < 2) {
                player.lane++;
                player.x = lanePositions[player.lane];
                keys['ArrowRight'] = false;
            }

            drawCar(player.x, player.y, player.width, player.height, carColor);

            spawnObstacle();
            for (let i = obstacles.length - 1; i >= 0; i--) {
                // Κάθε αυτοκίνητο κινείται με τη δική του ταχύτητα
                obstacles[i].y += (speed / 8) * obstacles[i].speedMultiplier;
                
                // Ζωγραφίζουμε ανάλογα με τον τύπο
                if (obstacles[i].type === 'truck') {
                    drawTruck(obstacles[i].x, obstacles[i].y, obstacles[i].width, obstacles[i].height, obstacles[i].color);
                } else if (obstacles[i].type === 'bus') {
                    drawBus(obstacles[i].x, obstacles[i].y, obstacles[i].width, obstacles[i].height, obstacles[i].color);
                } else {
                    drawCar(obstacles[i].x, obstacles[i].y, obstacles[i].width, obstacles[i].height, obstacles[i].color);
                }
                
                if (obstacles[i].y > canvas.height) {
                    obstacles.splice(i, 1);
                }
            }

            if (checkCollision()) {
                gameRunning = false;
                if (animationFrameId) {
                    cancelAnimationFrame(animationFrameId);
                }
                gameOverDiv.style.display = 'block';
                document.getElementById('finalDistance').textContent = Math.round(distance);
                document.getElementById('finalTime').textContent = `${minutes}:${seconds.toString().padStart(2, '0')}`;
                return;
            }

            animationFrameId = requestAnimationFrame(gameLoop);
        }

        document.addEventListener('keydown', (e) => {
            if (e.code === 'Space' && !gameRunning && gameOverDiv.style.display === 'block') {
                startGame();
            }
            if (e.code === 'ControlLeft') {
                if (gameRunning || gameOverDiv.style.display === 'block') {
                    gameRunning = false;
                    gameOverDiv.style.display = 'none';
                    startGame();
                }
            }
            if (e.code === 'Tab') {
                e.preventDefault();
                gameRunning = false;
                gameScreen.style.display = 'none';
                startScreen.style.display = 'block';
                gameOverDiv.style.display = 'none';
            }
            keys[e.key] = true;
        });

        document.addEventListener('keyup', (e) => {
            keys[e.key] = false;
        });
